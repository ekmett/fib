cmake_minimum_required(VERSION 3.3)
project(FIB)
add_compile_options(-std=c++11)

find_package(Boost 1.63.0 COMPONENTS context)

if(Boost_FOUND)

    add_library(fib worker.cpp) 
    target_link_libraries(fib ${Boost_LIBRARIES})
endif()
